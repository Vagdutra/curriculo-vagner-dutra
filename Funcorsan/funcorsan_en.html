contribution_audit_query.sql
SELECT p.participant_name, pl.plan_code, pl.enrollment_date, r.contribution_value, -- Using CASE for enhanced portability over DECODE CASE pl.status WHEN 'A' THEN 'Active' WHEN 'I' THEN 'Inactive' ELSE 'Unknown' END AS status_desc, r.reference_date AS reserve_competence FROM tb_participant p INNER JOIN tb_plan pl ON p.participant_id = pl.participant_id INNER JOIN tb_reserve r ON pl.plan_id = r.plan_id AND r.reference_date = (SELECT MAX(reference_date) FROM tb_reserve WHERE plan_id = pl.plan_id) WHERE pl.status = 'A' AND (r.contribution_value <= 0) ORDER BY pl.enrollment_date DESC;
